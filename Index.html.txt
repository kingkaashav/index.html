    /* Sentence builder (from original site) */
    let chosenFeeling = null;
    let chosenFood = null;

    function setFeeling(type) {
      chosenFeeling = type;
      updateSentencePreview();
    }

    function setFood(foodKey) {
      chosenFood = foodKey;
      updateSentencePreview();
    }

    function updateSentencePreview() {
      const box = document.getElementById("sentence-output");
      if (!chosenFeeling && !chosenFood) {
        box.innerHTML = "";
        return;
      }

      let text = "<div class='tip'>You’ve selected: ";
      if (chosenFeeling) {
        text += chosenFeeling === "like" ? "“I like”" : "“I don’t like”";
      }
      if (chosenFood) {
        text += chosenFeeling ? " + " : "";
        text += foods[chosenFood].english;
      }
      text += "</div>";
      box.innerHTML = text;
    }

    function buildSentence() {
      const box = document.getElementById("sentence-output");

      if (!chosenFeeling || !chosenFood) {
        box.innerHTML = `
          <div class="tip">Choose both a feeling (like / don’t like) and a food first.</div>
        `;
        return;
      }

      const foodWord = foods[chosenFood].gujarati;
      let gujaratiSentence = "";
      let englishSentence = "";

      if (chosenFeeling === "like") {
        gujaratiSentence = `મને ${foodWord} ગમે છે`;
        englishSentence = `I like ${foods[chosenFood].english}.`;
      } else {
        gujaratiSentence = `મને ${foodWord} ગમતું નથી`;
        englishSentence = `I don’t like ${foods[chosenFood].english}.`;
      }

      box.innerHTML = `
        <div class="phrase">${gujaratiSentence}</div>
        <div class="translation">${englishSentence}</div>
        <div class="tip">Try saying it out loud!</div>
      `;
    }
